🧠 AuthController
bash
php artisan make:controller AuthController
phpapp/Http/Controllers/AuthController.php
public function register(Request $request)
{
    $request->validate([
        'name'  => 'required|string|max:255',
        'email' => 'required|string|email|unique:users',
        'password' => 'required|string|min:8',
    ]);

    $user = User::create([
        'name'     => $request->name,
        'email'    => $request->email,
        'password' => bcrypt($request->password),
        'role'     => 'user', // always 'user', never trust client
    ]);

    $token = $user->createToken('auth_token')->plainTextToken;

    return response()->json([
        'user'  => $user,
        'token' => $token,
    ], 201);
}

public function login(Request $request)
{
    if (!Auth::attempt($request->only('email', 'password'))) {
        return response()->json([
            'message' => 'Invalid credentials'
        ], 401);
    }

    $user  = Auth::user();
    $token = $user->createToken('auth_token')->plainTextToken;

    return response()->json([
        'user'  => $user,
        'token' => $token,
    ]);
}

public function logout(Request $request)
{
    $request->user()->currentAccessToken()->delete();
    return response()->json(['message' => 'Logged out']);
}

public function me(Request $request)
{
    return response()->json($request->user());
}
✅ Security FixThis completely removes the "admin email hack". Roles are set only in the backend. To make someone admin, manually update the DB: UPDATE users SET role='admin' WHERE email='you@example.com';

🔍 BookController — Index with Filters
phpBookController.php
public function index(Request $request)
{
    $query = Book::query();

    // Filter by genre: /api/books?genre=Fantasy
    if ($request->filled('genre')) {
        $query->where('genre', $request->genre);
    }

    // Search by title or author: /api/books?search=tolkien
    if ($request->filled('search')) {
        $query->where(function($q) use ($request) {
            $q->where('title', 'like', "%{$request->search}%")
              ->orWhere('author', 'like', "%{$request->search}%");
        });
    }

    // Filter featured/deals: /api/books?featured=1
    if ($request->boolean('featured')) {
        $query->where('is_featured', true);
    }

    if ($request->boolean('deals')) {
        $query->where('is_deal', true);
    }

    return response()->json(
        $query->paginate(20) // Fixes "no pagination" issue!
    );
}

🧠 CartController — Key Methods
phpCartController.php
public function index(Request $request)
{
    $cart = $request->user()->cart()->with('items.book')->first();
    return response()->json($cart);
}

public function add(Request $request)
{
    $request->validate(['book_id' => 'required|exists:books,id']);

    // Get or create the user's cart
    $cart = $request->user()->cart()->firstOrCreate();

    // If book already in cart, increment qty; otherwise add
    $item = $cart->items()->where('book_id', $request->book_id)->first();

    if ($item) {
        $item->increment('quantity');
    } else {
        $cart->items()->create([
            'book_id'  => $request->book_id,
            'quantity' => 1,
        ]);
    }

    return response()->json(['message' => 'Added to cart']);
}

public function update(Request $request, $itemId)
{
    $item = CartItem::findOrFail($itemId);
    $item->update(['quantity' => $request->quantity]);
    return response()->json(['message' => 'Updated']);
}
❤️ WishlistController — Toggle Logic
phpWishlistController.php
public function toggle(Request $request, $bookId)
{
    $user = $request->user();
    $existing = $user->wishlists()
                      ->where('book_id', $bookId)
                      ->first();

    if ($existing) {
        $existing->delete();
        return response()->json(['wishlisted' => false]);
    }

    $user->wishlists()->create(['book_id' => $bookId]);
    return response()->json(['wishlisted' => true]);
}

🧾 OrderController — Place Order
phpOrderController.php
public function store(Request $request)
{
    $user = $request->user();
    $cart = $user->cart()->with('items.book')->first();

    if (!$cart || $cart->items->isEmpty()) {
        return response()->json(['message' => 'Cart is empty'], 400);
    }

    // Calculate total
    $total = $cart->items->sum(fn($item) =>
        $item->book->price * $item->quantity
    );

    // Create order
    $order = Order::create([
        'user_id'          => $user->id,
        'total_amount'     => $total,
        'shipping_address' => $request->shipping_address,
        'status'           => 'pending',
    ]);

    // Copy cart items into order items
    foreach ($cart->items as $item) {
        $order->items()->create([
            'book_id'    => $item->book_id,
            'quantity'   => $item->quantity,
            'unit_price' => $item->book->price,
        ]);
    }

    // Clear the cart after ordering
    $cart->items()->delete();

    return response()->json([
        'message' => 'Order placed successfully',
        'order'   => $order->load('items.book'),
    ], 201);
}
💡 Payment IntegrationFor payment, integrate Razorpay after checkout: create a Razorpay order on POST /orders, return the razorpay_order_id to React, open the Razorpay checkout modal, then verify the payment signature on a POST /orders/{id}/verify-payment endpoint before marking the order as confirmed.

🔐 Update useAuthStore (Zustand)

import api from '../lib/api';

export const useAuthStore = create(persist((set) => ({
  user: null,
  token: null,

  login: async (email, password) => {
    const { data } = await api.post('/login', { email, password });
    localStorage.setItem('auth_token', data.token);
    set({ user: data.user, token: data.token });
  },

  register: async (name, email, password) => {
    const { data } = await api.post('/register', { name, email, password });
    localStorage.setItem('auth_token', data.token);
    set({ user: data.user, token: data.token });
  },

  logout: async () => {
    await api.post('/logout');
    localStorage.removeItem('auth_token');
    set({ user: null, token: null });
  },

  isAdmin: () => useAuthStore.getState().user?.role === 'admin',
}), { name: 'auth-store' }));
